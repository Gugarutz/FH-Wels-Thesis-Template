\makeatletter
\renewcommand*{\listoftables}{%
	\begingroup
	\let\oldaddchaptertocentry\addchaptertocentry
	\renewcommand*{\addchaptertocentry}[2]{\oldaddchaptertocentry{\thechapter}{##2}}%
	\KOMAoptions{open=any}%
	\clearpage
	\refstepcounter{chapter}%
	{\let\ps@plain\ps@empty
		\chapter[\listtablename]{\thechapter\hskip1ex\listtablename}}%
	\@starttoc{lot}%
	\endgroup
}
\makeatother
