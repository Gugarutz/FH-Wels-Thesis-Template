    \AtBeginDocument{\RenewCommandCopy\set\qty}

    \AtBeginDocument{\RenewCommandCopy\qty\SI}

% silences the warning that \qty is overwritten see:

    \ExplSyntaxOn
    \msg_redirect_name:nnn { siunitx } { physics-pkg } { none }
    \ExplSyntaxOff
