% Configures references, citations, hyperlinks, and captions.

%-------------------------------------------------------------------------------%
%                          REFERENCING AND HYPERLINKING                         %
%-------------------------------------------------------------------------------%

        \usepackage{caption}                % Better caption control
        \usepackage{subcaption}             % Subfigures, subtables
        \usepackage[hidelinks]{hyperref}    % Clickable references without colored boxes
        \usepackage{hyperref-patches}       % Compatibility patches
        \usepackage{cleveref}               % Smart cross-referencing

%-------------------------------------------------------------------------------%
%                                 BIBLIOGRAPHY                                  %
%-------------------------------------------------------------------------------%

    \usepackage[
        backend=biber,
        maxcitenames=3,
        sorting=none
    ]{biblatex}

    % Font size for bibliography
    \renewcommand{\bibfont}{\small}  % Adjust size here

%-------------------------------------------------------------------------------%
%                              CAPTION CONFIGURATION                            %
%-------------------------------------------------------------------------------%

    \captionsetup{
        font=small,
        labelfont=bf,
        aboveskip=10pt,
        belowskip=5pt,
        margin=5mm,
        justification=raggedright
    }

%-------------------------------------------------------------------------------%
%                            NUMBERING IN CHAPTER CONTEXT                       %
%-------------------------------------------------------------------------------%

    \counterwithin{figure}  {chapter}   % Figure numbering includes chapter
    \counterwithin{table}   {chapter}   % Table numbering includes chapter
    \counterwithin{equation}{chapter}   % Equation numbering includes chapter

    % Same but for sections, not tested

%-------------------------------------------------------------------------------%
%                           LANGUAGE-SPECIFIC LABEL NAMES                       %
%-------------------------------------------------------------------------------%

    % Figure captions
    \addto\captionsngerman  {\renewcommand{\figurename}{Abb.}}
    \addto\captionsenglish  {\renewcommand{\figurename}{Fig.}}

    % Table captions
    \addto\captionsngerman  {\renewcommand{\tablename}{Tab.}}
    \addto\captionsenglish  {\renewcommand{\tablename}{Tab.}}
