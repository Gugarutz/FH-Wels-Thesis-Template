\usepackage{listings}	% Base package to display code snippets

%------------------------------------------------------------%
% STYLE DEFINITIONS
%------------------------------------------------------------%

% Python Style
\lstdefinestyle{pythonstyle}{
	language=Python,
	basicstyle=\ttfamily\scriptsize,
	keywordstyle=\color{blue}\bfseries,
	commentstyle=\color{gray},
	stringstyle=\color{red},
	numbers=left,
	numberstyle=\tiny\color{gray}\ttfamily,
	stepnumber=5,        % Number every 5th line
	firstnumber=1,       % Start numbering at 1
	numbersep=5pt,       % Adjust spacing
	frame=single,
	showstringspaces=false,
	xleftmargin=20pt,
	xrightmargin=10pt,
	tabsize=2,  % Decrease the tab size (default is 8)
	breaklines=true, % Enable line wrapping
}
