	\newcommand{\setSectioning}%
	{%
		\RedeclareSectionCommand[%
			tocnumwidth	= 4em,	% Width reserved for the section number
			indent		= 0pt,%
			beforeskip	= -3.5ex, 	% Adjust this value as needed
			afterskip	= 2.3ex,%
			tocindent	= 1.5em]%
		{section}%
		\RedeclareSectionCommand[%
			tocnumwidth	= 4em,	% Width reserved for the section number
			indent		= 0pt,%
			beforeskip	= 3.5ex, 	% Adjust this value as needed
			afterskip	= 0.1ex,	%
			tocindent	= 1.5em]%
		{subsection}%
		\RedeclareSectionCommand[%
			tocnumwidth	= 4em,	% Width reserved for the section number
			indent		= 0pt,%
			beforeskip	= -3.25ex, % Adjust this value as needed
			afterskip	= 0.1ex,%
			tocindent	= 1.5em]%
		{subsubsection}%
		\RedeclareSectionCommand[%
			tocnumwidth	= 4em,	% Width reserved for the section number
			indent		= 0pt,%
			beforeskip	= -3.25ex, % Adjust this value as needed
			afterskip	= 0.1ex,%
			tocindent	= 1.5em]%
		{paragraph}%
		\RedeclareSectionCommand[%
			tocnumwidth	= 4em,	% Width reserved for the section number
			indent		= 0pt,%
			beforeskip	=-3.25ex, % Adjust this value as needed
			afterskip	= 0.1ex,%
			tocindent	= 1.5em]%
		{subparagraph}%
	}
