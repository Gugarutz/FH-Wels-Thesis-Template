%------------------------------------------------------------%
% GENERAL GRAPHICS AND FLOAT MANAGEMENT

    \usepackage{graphicx}   		% Graphics package, easily include  images (see Insert Graphic Wizard)
    \usepackage[x11names]{xcolor}   % Sophistacated package for colours, with table option to use colors in tables
    \usepackage{float}				% more control for placing floats (tables, figures,...)

    \usepackage{wrapfig}    		% allows wrapping text around an image
    \usepackage{rotating}
    \usepackage{pdfpages}   		% allows inclusion of pdfs

%------------------------------------------------------------%
% TABLES
%------------------------------------------------------------%

	\usepackage{array}			% Provides advanced table formatting options
	\usepackage{longtable}		% Allows tables to span multiple pages
	\usepackage{booktabs}		% Provides professional-looking table rules
	\usepackage{colortbl}		% Enables colored table cells
	\usepackage{multicol}		% Allows multi-column cells in tables
	\usepackage{multirow}		% Allows multi-row cells in tables
	\usepackage{nicematrix}		% custom table rules and decorations
    \usepackage{tabularx}       % equal distribution of columns
    \usepackage{tabulary}       % expanded version of tabularx
